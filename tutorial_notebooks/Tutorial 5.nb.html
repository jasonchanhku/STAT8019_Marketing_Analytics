imp<-data.frame(scale(t(fit.conj$imp)))
dist<-dist(imp,method="euclidean")^2
clust <- hclust(dist, method="ward.D")

ggplot(mapping=aes(x=1:length(clust$height),y=clust$height))+
  geom_line()+
  geom_point()+
  labs(x="stage",y="height")

par(mar=c(1,4,1,1))
